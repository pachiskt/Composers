class CuentaBancaria:
    def __init__(self, saldo: float, tasa_anual: float):
        self.saldo = saldo
        self.num_depositos = 0
        self.num_retiros = 0
        self.tasa_anual = tasa_anual
        self.comision_mensual = 0.0
    
    def depositar(self, cantidad: float):
        if cantidad <= 0:
            print("‚úó Error: La cantidad debe ser positiva")
            return False
        
        # Validar que no exceda l√≠mites razonables
        if cantidad > 1000000:  # L√≠mite de $1,000,000 por dep√≥sito
            print("‚úó Error: El dep√≥sito no puede exceder $1,000,000")
            return False
        
        self.saldo += cantidad
        self.num_depositos += 1
        print(f"‚úì Dep√≥sito exitoso: ${cantidad:.2f}")
        print(f"  Saldo actual: ${self.saldo:.2f}")
        return True
    
    def retirar(self, cantidad: float):
        if cantidad <= 0:
            print("‚úó Error: La cantidad debe ser positiva")
            return False
        
        # Validar que no exceda l√≠mites razonables
        if cantidad > 500000:  # L√≠mite de $500,000 por retiro
            print("‚úó Error: El retiro no puede exceder $500,000")
            return False
        
        return True
    
    def calcular_interes_mensual(self):
        if self.saldo > 0:
            interes_mensual = (self.tasa_anual / 12) / 100
            interes = self.saldo * interes_mensual
            self.saldo += interes
            print(f"‚úì Inter√©s mensual: ${interes:.2f}")
            return interes
        return 0
    
    def extracto_mensual(self):
        # Aplicar comisi√≥n mensual
        if self.comision_mensual > 0:
            self.saldo -= self.comision_mensual
            print(f"‚úì Comisi√≥n mensual aplicada: ${self.comision_mensual:.2f}")
        # Calcular inter√©s
        self.calcular_interes_mensual()
    
    def imprimir(self):
        print(f"Saldo: ${self.saldo:.2f}")
        print(f"Dep√≥sitos: {self.num_depositos}")
        print(f"Retiros: {self.num_retiros}")
        print(f"Tasa anual: {self.tasa_anual}%")
        print(f"Comisi√≥n mensual: ${self.comision_mensual:.2f}")


class CuentaAhorros(CuentaBancaria):
    def __init__(self, saldo: float, tasa_anual: float):
        super().__init__(saldo, tasa_anual)
        self.activa = saldo >= 10000
        self.comision_mensual = 0.0
    
    def depositar(self, cantidad: float):
        # En Cuenta de Ahorros SI se permiten dep√≥sitos incluso estando inactiva
        # porque pueden activar la cuenta
        
        # Validar que el saldo resultante no sea excesivo
        saldo_futuro = self.saldo + cantidad
        if saldo_futuro > 50000000:  # L√≠mite de $50,000,000
            print("‚úó Error: El saldo no puede exceder $50,000,000")
            return False
        
        estado_anterior = self.activa
        
        if super().depositar(cantidad):
            # Verificar si se activa la cuenta despu√©s del dep√≥sito
            self.activa = self.saldo >= 10000
            
            if not estado_anterior and self.activa:
                print("üéâ ¬°Cuenta ACTIVADA! (Saldo ‚â• $10,000)")
            elif estado_anterior and not self.activa:
                print("‚ö† Cuenta DESACTIVADA (Saldo < $10,000)")
                
            return True
        return False
    
    def retirar(self, cantidad: float):
        # En Cuenta de Ahorros NO se permiten retiros estando inactiva
        if not self.activa:
            print("‚úó Cuenta INACTIVA - No se permiten retiros")
            print("  Para activar la cuenta, el saldo debe ser ‚â• $10,000")
            return False
        
        # Validar que no genere saldo negativo
        if cantidad > self.saldo:
            print("‚úó Fondos insuficientes")
            print(f"  Saldo disponible: ${self.saldo:.2f}")
            return False
        
        estado_anterior = self.activa
        
        if super().retirar(cantidad):
            self.saldo -= cantidad
            self.num_retiros += 1
            print(f"‚úì Retiro exitoso: ${cantidad:.2f}")
            print(f"  Saldo actual: ${self.saldo:.2f}")
            
            # Verificar si se mantiene activa despu√©s del retiro
            self.activa = self.saldo >= 10000
            
            if estado_anterior and not self.activa:
                print("‚ö† Cuenta ahora est√° INACTIVA (saldo < $10,000)")
            return True
        return False
    
    def extracto_mensual(self):
        # Comisi√≥n por exceso de retiros (m√°s de 3 retiros)
        if self.num_retiros > 3:
            comision_extra = (self.num_retiros - 3) * 1000
            self.comision_mensual += comision_extra
            print(f"‚ö† Comisi√≥n por exceso de retiros: ${comision_extra:.2f}")
        
        super().extracto_mensual()
        
        # Validar que no haya saldo negativo despu√©s de comisiones
        if self.saldo < 0:
            print("‚ö† Ajuste: El saldo no puede ser negativo, se establece en $0.00")
            self.saldo = 0
        
        # Actualizar estado de la cuenta
        estado_anterior = self.activa
        self.activa = self.saldo >= 10000
        
        if not estado_anterior and self.activa:
            print("üéâ ¬°Cuenta ACTIVADA por inter√©s/comisiones! (Saldo ‚â• $10,000)")
        elif estado_anterior and not self.activa:
            print("‚ö† Cuenta DESACTIVADA por inter√©s/comisiones (Saldo < $10,000)")
    
    def imprimir(self):
        print("\n" + "="*50)
        print("CUENTA DE AHORROS")
        print("="*50)
        print(f"Saldo: ${self.saldo:.2f}")
        print(f"Comisi√≥n mensual: ${self.comision_mensual:.2f}")
        print(f"Total transacciones: {self.num_depositos + self.num_retiros}")
        print(f"Estado: {'ACTIVA' if self.activa else 'INACTIVA'}")
        print(f"Condici√≥n: {'Saldo ‚â• $10,000' if self.activa else 'Saldo < $10,000'}")


class CuentaCorriente(CuentaBancaria):
    def __init__(self, saldo: float, tasa_anual: float):
        super().__init__(saldo, tasa_anual)
        self.sobregiro = 0.0
        self.comision_mensual = 10000  # Comisi√≥n fija mensual
    
    def retirar(self, cantidad: float):
        if cantidad <= 0:
            print("‚úó Error: La cantidad debe ser positiva")
            return False
        
        # Validar l√≠mite de retiro
        if cantidad > 500000:
            print("‚úó Error: El retiro no puede exceder $500,000")
            return False
        
        # Validar l√≠mite total (saldo + sobregiro disponible)
        limite_total = self.saldo + 100000  # $100,000 de sobregiro m√°ximo
        if cantidad > limite_total:
            print("‚úó Error: Excede el l√≠mite disponible")
            print(f"  L√≠mite total: ${limite_total:.2f}")
            return False
        
        if cantidad > self.saldo:
            # Calcular sobregiro
            self.sobregiro = cantidad - self.saldo
            self.saldo = 0
            self.num_retiros += 1
            print(f"‚úì Retiro con SOBREGIRRO: ${cantidad:.2f}")
            print(f"  Sobregiro utilizado: ${self.sobregiro:.2f}")
            print(f"  Saldo actual: $0.00")
            return True
        else:
            self.saldo -= cantidad
            self.num_retiros += 1
            print(f"‚úì Retiro exitoso: ${cantidad:.2f}")
            print(f"  Saldo actual: ${self.saldo:.2f}")
            return True
    
    def depositar(self, cantidad: float):
        if cantidad <= 0:
            print("‚úó Error: La cantidad debe ser positiva")
            return False
        
        # Validar l√≠mite de dep√≥sito
        if cantidad > 1000000:
            print("‚úó Error: El dep√≥sito no puede exceder $1,000,000")
            return False
        
        if cantidad > 0:
            if self.sobregiro > 0:
                # Primero cubrir el sobregiro
                if cantidad <= self.sobregiro:
                    self.sobregiro -= cantidad
                    print(f"‚úì Dep√≥sito aplicado al SOBREGIRRO: ${cantidad:.2f}")
                    print(f"  Sobregiro restante: ${self.sobregiro:.2f}")
                else:
                    # Cubrir sobregiro y el resto al saldo
                    excedente = cantidad - self.sobregiro
                    self.saldo += excedente
                    self.sobregiro = 0
                    print(f"‚úì Dep√≥sito exitoso: ${cantidad:.2f}")
                    print(f"  Sobregiro cubierto, saldo actual: ${self.saldo:.2f}")
            else:
                self.saldo += cantidad
                print(f"‚úì Dep√≥sito exitoso: ${cantidad:.2f}")
                print(f"  Saldo actual: ${self.saldo:.2f}")
            
            self.num_depositos += 1
            return True
        else:
            print("‚úó Error: La cantidad debe ser positiva")
            return False
    
    def extracto_mensual(self):
        super().extracto_mensual()
        
        # Validar que no haya saldo negativo excesivo despu√©s de comisiones
        if self.saldo < -100000:  # L√≠mite de sobregiro
            print("‚ö† Ajuste: Sobregiro excede l√≠mite, se establece en -$100,000")
            self.saldo = -100000
        
        print(f"  Saldo final: ${self.saldo:.2f}")
    
    def imprimir(self):
        print("\n" + "="*50)
        print("CUENTA CORRIENTE")
        print("="*50)
        print(f"Saldo: ${self.saldo:.2f}")
        print(f"Comisi√≥n mensual: ${self.comision_mensual:.2f}")
        print(f"Total transacciones: {self.num_depositos + self.num_retiros}")
        print(f"Sobregiro: ${self.sobregiro:.2f}")
        if self.saldo < 0:
            print("‚ö† CUENTA EN SOBREGIRRO")


def obtener_numero_valido(mensaje, min_val=0.01, max_val=100000000):
    """Valida que la entrada sea un n√∫mero v√°lido"""
    while True:
        try:
            valor = float(input(mensaje))
            if valor < min_val:
                print(f"‚úó Error: El valor debe ser mayor o igual a ${min_val:.2f}")
            elif valor > max_val:
                print(f"‚úó Error: El valor no puede exceder ${max_val:.2f}")
            else:
                return valor
        except ValueError:
            print("‚úó Error: Por favor ingrese un n√∫mero v√°lido")
